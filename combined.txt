function loadDoc(option){

function iterClick(jsonResponse){
	

	var i;
	var tbl = document.getElementById("tblBlank");
	
	var txt = "";
	
	for(i = 0; i< jsonResponse.pets.length; i++){
		txt += "<tr><td>";
		txt += jsonResponse.pets[i].animal;
		//txt += "hi";
		txt += "</td></tr>";
	}
	
	tbl.innerHTML = txt;
	
}
